## 一、文件读取（file_reader.py）

### 统一读取接口

```python
import pandas as pd
from pathlib import Path

def read(file_path: str) -> pd.DataFrame:
    """统一读取入口：自动识别格式并读取为 DataFrame。

    支持格式：.xlsx, .xls, .csv
    CSV 自动检测编码。
    """
    ext = Path(file_path).suffix.lower()

    if ext == '.csv':
        return _read_csv(file_path)
    elif ext in ('.xlsx', '.xlsm'):
        return _read_xlsx(file_path)
    elif ext == '.xls':
        return _read_xls(file_path)
    else:
        raise ValueError(f"不支持的文件格式: {ext}")
```

### CSV 读取（含编码检测）

```python
import chardet

def _read_csv(file_path: str) -> pd.DataFrame:
    # 1. 检测编码
    with open(file_path, 'rb') as f:
        raw = f.read(100000)  # 读前 100KB 用于检测
    detected = chardet.detect(raw)
    encoding = detected['encoding'] or 'utf-8'
    confidence = detected['confidence']

    # 2. 尝试顺序：检测到的编码 → utf-8 → gbk → gb2312 → latin-1
    encodings_to_try = [encoding]
    if encoding.lower() not in ('utf-8', 'utf8'):
        encodings_to_try.extend(['utf-8', 'gbk', 'gb2312'])
    encodings_to_try.append('latin-1')  # 兜底，不会报错

    for enc in encodings_to_try:
        try:
            df = pd.read_csv(file_path, encoding=enc, dtype=str)
            return df
        except (UnicodeDecodeError, UnicodeError):
            continue

    raise ValueError(f"无法解码 CSV 文件，尝试了: {encodings_to_try}")
```

### xlsx/xls 读取

```python
def _read_xlsx(file_path: str) -> pd.DataFrame:
    # openpyxl 引擎，dtype=str 防止自动类型推断导致数据丢失
    return pd.read_excel(file_path, engine='openpyxl', dtype=str)

def _read_xls(file_path: str) -> pd.DataFrame:
    # xlrd 引擎处理老旧 .xls 格式
    return pd.read_excel(file_path, engine='xlrd', dtype=str)
```

注意：所有格式统一用 `dtype=str` 读取，避免 Pandas 自动将 `00123` 推断为 `123`。类型转换在 Agent 生成的代码中按需执行。

## 二、Schema 提取（schema.py）

```python
import pandas as pd
import numpy as np

def extract(df: pd.DataFrame) -> dict:
    """从 DataFrame 提取 Schema 信息。

    返回:
    {
        "columns": ["姓名", "年龄", "部门"],
        "dtypes": {"姓名": "object", "年龄": "int64", "部门": "object"},
        "row_count": 500,
        "null_counts": {"姓名": 0, "年龄": 3, "部门": 0},
        "sample": [{"姓名": "张三", "年龄": "28", "部门": "研发部"}, ...],
        "unique_counts": {"姓名": 495, "年龄": 45, "部门": 8},  # 高基数 vs 低基数列
        "numeric_columns": ["年龄"],
        "date_candidate_columns": []  # 可能包含日期的列
    }
    """
    # 基础 schema
    schema = {
        "columns": list(df.columns),
        "dtypes": {col: str(dtype) for col, dtype in df.dtypes.items()},
        "row_count": len(df),
        "null_counts": {col: int(df[col].isna().sum()) for col in df.columns},
    }

    # 样本数据（取前 5 行，处理 NaN）
    sample_df = df.head(5).replace({np.nan: None, pd.NaT: None})
    schema["sample"] = sample_df.to_dict(orient="records")

    # 唯一值数量（帮助 LLM 理解列的含义）
    schema["unique_counts"] = {
        col: int(df[col].nunique()) for col in df.columns
    }

    # 尝试识别数值列
    schema["numeric_columns"] = [
        col for col in df.columns
        if pd.to_numeric(df[col], errors='coerce').notna().sum() > len(df) * 0.8
    ]

    # 尝试识别日期候选列
    schema["date_candidate_columns"] = [
        col for col in df.columns
        if _looks_like_date(df[col])
    ]

    return schema


def _looks_like_date(series: pd.Series) -> bool:
    """检查一个列是否可能包含日期数据。"""
    sample = series.dropna().head(20)
    if len(sample) == 0:
        return False
    success = 0
    for val in sample:
        try:
            pd.to_datetime(str(val))
            success += 1
        except (ValueError, TypeError):
            pass
    return success / len(sample) > 0.6  # 60% 以上的值可解析为日期
```

## 三、文件写入与格式降级（file_writer.py）

### 统一写入接口

```python
import pandas as pd
from pathlib import Path

def write(df: pd.DataFrame, target_spec: dict, output_dir: str = "data/outputs") -> str:
    """根据目标规格写出文件。

    target_spec 包含:
        - target_format: "xlsx" | "xls" | "csv"
        - target_encoding: "utf-8" | "gbk" | "gb2312" (csv 时)
        - task_id: 用于生成文件名

    返回输出文件路径。
    """
    fmt = target_spec["target_format"]
    task_id = target_spec.get("task_id", "output")
    output_path = Path(output_dir) / f"{task_id}_converted.{fmt}"

    Path(output_dir).mkdir(parents=True, exist_ok=True)

    if fmt == "xlsx":
        return _write_xlsx(df, str(output_path))
    elif fmt == "xls":
        return _write_xls(df, str(output_path))
    elif fmt == "csv":
        encoding = target_spec.get("target_encoding", "utf-8")
        return _write_csv(df, str(output_path), encoding)
    else:
        raise ValueError(f"不支持的目标格式: {fmt}")
```

### xls 写入（格式降级核心）

```python
def _write_xls(df: pd.DataFrame, path: str) -> str:
    """写入 Excel 97-2003 .xls 格式。

    兼容性处理：
    1. 行数限制：xls 最多 65536 行，超出截断并警告
    2. 列数限制：xls 最多 256 列，超出截断
    3. 单元格字符限制：单个单元格最多 32767 字符
    """
    if len(df) > 65536:
        # 截断 + 记录警告（此信息应返回给前端）
        df = df.head(65536)
        # 在第一列名后追加警告标记
        # 返回的 path 中也可以附带 warning 信息

    if len(df.columns) > 256:
        df = df.iloc[:, :256]

    # 截断过长单元格内容
    for col in df.columns:
        df[col] = df[col].apply(
            lambda x: str(x)[:32767] if pd.notna(x) and len(str(x)) > 32767 else x
        )

    # xlwt 引擎写 .xls
    with pd.ExcelWriter(path, engine='xlwt') as writer:
        df.to_excel(writer, index=False, sheet_name='Sheet1')
    return path
```

### CSV 写入（编码兼容核心）

```python
def _write_csv(df: pd.DataFrame, path: str, encoding: str) -> str:
    """写入 CSV。

    兼容性处理（针对老旧系统）：
    1. GBK/GB2312 编码：写入 BOM 头（`﻿` 或 `\xef\xbb\xbf`）
    2. 日期格式：统一为 YYYYMMDD 无分隔符格式（常见于银行/ERP 老旧系统）
    3. 换行符：使用 \r\n (Windows 标准)
    """

    # 如果是 GBK 编码，某些系统需要 UTF-8 BOM 来识别
    if encoding.lower() in ('gbk', 'gb2312'):
        # GBK 不需要 BOM，但需要确保文件以 GBK 编码写入
        pass

    df.to_csv(
        path,
        index=False,
        encoding=encoding,
        line_terminator='\r\n',
        errors='replace',  # 无法编码的字符替换为 ?
    )

    # 如果指定了 UTF-8 且需要 BOM
    if encoding.lower() in ('utf-8', 'utf8'):
        _prepend_bom(path)

    return path


def _prepend_bom(path: str):
    """在 UTF-8 文件开头添加 BOM。"""
    with open(path, 'r+b') as f:
        content = f.read()
        f.seek(0)
        f.write(b'\xef\xbb\xbf' + content)
```

### 写入结果应该包含警告信息

```python
def write_with_warnings(df: pd.DataFrame, target_spec: dict) -> dict:
    """写文件并返回结果，包含兼容性警告。"""
    warnings = []
    fmt = target_spec["target_format"]

    # xls 行数检查
    if fmt == "xls" and len(df) > 65536:
        truncated_rows = len(df) - 65536
        warnings.append(f"xls 格式最多支持 65536 行，已截断最后 {truncated_rows} 行")
        df = df.head(65536)

    # xls 列数检查
    if fmt == "xls" and len(df.columns) > 256:
        truncated_cols = len(df.columns) - 256
        warnings.append(f"xls 格式最多支持 256 列，已截断最后 {truncated_cols} 列")
        df = df.iloc[:, :256]

    file_path = write(df, target_spec)

    return {
        "file_path": file_path,
        "warnings": warnings,
    }
```

## 四、Diff 计算（diff.py）

```python
import pandas as pd
import numpy as np

def compute(original: pd.DataFrame, transformed: pd.DataFrame) -> dict:
    """计算两个 DataFrame 之间的差异。

    返回前端可直接渲染的 Diff 数据：
    {
        "row_counts": {"original": 500, "transformed": 488},
        "added_columns": ["省", "市", "区"],
        "removed_columns": ["地址"],
        "renamed_columns": {"姓名": "Emp_Name", "年龄": "Age"},
        "added_rows": [0, 1, 2],        # 新增行的原索引
        "removed_rows": [12, 45, 78],   # 删除行的原索引
        "modified_cells": [
            {"row": 3, "col": "电话", "old": "010-1234-5678", "new": "01012345678"},
            {"row": 7, "col": "金额", "old": "-100", "new": None},  # None 表示被删除
        ],
        "unchanged_row_count": 476,
    }
    """

    result = {
        "row_counts": {"original": len(original), "transformed": len(transformed)},
        "added_columns": [],
        "removed_columns": [],
        "renamed_columns": {},
        "added_rows": [],
        "removed_rows": [],
        "modified_cells": [],
        "unchanged_row_count": 0,
    }

    orig_cols = set(original.columns)
    trans_cols = set(transformed.columns)

    result["added_columns"] = list(trans_cols - orig_cols)
    result["removed_columns"] = list(orig_cols - trans_cols)

    # 列重命名检测：共同列数 = 0，但列数量相同 → 可能是重命名
    common_cols = list(orig_cols & trans_cols)

    # 逐行比较（仅比较共同列）
    # 策略：基于行索引对齐
    orig_indexed = original.reset_index(drop=True)
    trans_indexed = transformed.reset_index(drop=True)

    # 行数变化
    if len(trans_indexed) < len(orig_indexed):
        # 有行被删除：逐行匹配找到被删除的行
        result["removed_rows"], result["modified_cells"], result["unchanged_row_count"] = \
            _row_diff(orig_indexed, trans_indexed, common_cols)
    elif len(trans_indexed) > len(orig_indexed):
        result["added_rows"] = list(range(len(orig_indexed), len(trans_indexed)))
    else:
        result["removed_rows"], result["modified_cells"], result["unchanged_row_count"] = \
            _row_diff(orig_indexed, trans_indexed, common_cols)

    return result


def _row_diff(orig: pd.DataFrame, trans: pd.DataFrame, cols: list[str]) -> tuple:
    """逐行比较共同列，找出删除和修改。"""
    removed_rows = []
    modified_cells = []
    unchanged = 0

    min_len = min(len(orig), len(trans))

    for i in range(min_len):
        row_changed = False
        for col in cols:
            if col not in orig.columns or col not in trans.columns:
                continue
            old_val = orig.at[i, col] if col in orig.columns else None
            new_val = trans.at[i, col] if col in trans.columns else None
            # NaN 统一处理：pd.isna(None) → True
            if pd.isna(old_val) and pd.isna(new_val):
                continue
            if str(old_val) != str(new_val):
                modified_cells.append({
                    "row": i,
                    "col": col,
                    "old": None if pd.isna(old_val) else old_val,
                    "new": None if pd.isna(new_val) else new_val,
                })
                row_changed = True
        if not row_changed:
            unchanged += 1

    # 超出 min_len 的行视为删除
    if len(orig) > len(trans):
        removed_rows = list(range(min_len, len(orig)))

    return removed_rows, modified_cells, unchanged
```

## 五、前端数据预览格式

前端预览表格需要的格式：

```json
{
  "columns": ["姓名", "年龄", "部门"],
  "rows": [
    ["张三", "28", "研发部"],
    ["李四", "35", "市场部"]
  ]
}
```

后端在返回 preview 时直接转为 `{columns: [...], rows: [[...], ...]}` 格式，前端无需处理 DataFrame 对象。
