## 一、SQLite 数据库设计（database.py）

### 表结构

```sql
-- 转换任务历史
CREATE TABLE conversion_tasks (
    task_id        TEXT PRIMARY KEY,
    source_filename TEXT NOT NULL,
    source_file_path TEXT,
    target_format  TEXT NOT NULL,
    instructions   TEXT,
    skill_id       TEXT,              -- 如果使用了已有技能
    status         TEXT NOT NULL,     -- 'completed' | 'failed' | 'cancelled'
    source_schema  TEXT,              -- JSON: {columns, dtypes, row_count}
    target_schema  TEXT,              -- JSON: {columns, format}
    execution_code TEXT,              -- 最终执行的 Pandas 代码
    error_message  TEXT,
    token_used     INTEGER,
    created_at     TEXT NOT NULL,     -- ISO8601
    completed_at   TEXT
);

-- 技能模板
CREATE TABLE skill_templates (
    skill_id       TEXT PRIMARY KEY,
    name           TEXT NOT NULL,
    description    TEXT,
    source_schema  TEXT NOT NULL,     -- JSON: 源 Schema 特征
    target_spec    TEXT NOT NULL,     -- JSON: 目标格式规格
    code           TEXT NOT NULL,     -- 可复用的 Pandas 转换代码
    column_mapping TEXT,              -- JSON: 列映射关系
    usage_count    INTEGER DEFAULT 0,
    created_at     TEXT NOT NULL,
    updated_at     TEXT NOT NULL
);

-- 创建索引
CREATE INDEX idx_tasks_created ON conversion_tasks(created_at);
CREATE INDEX idx_skills_name ON skill_templates(name);
CREATE INDEX idx_skills_usage ON skill_templates(usage_count DESC);
```

### database.py 核心接口

```python
import sqlite3
import uuid
import json
from datetime import datetime, timezone

DB_PATH = "data/morphsheet.db"

def get_connection() -> sqlite3.Connection:
    conn = sqlite3.connect(DB_PATH)
    conn.row_factory = sqlite3.Row
    return conn

def init_db():
    """初始化数据库，创建表。启动时调用一次。"""
    with get_connection() as conn:
        conn.executescript(SCHEMA_SQL)

# --- 任务历史 ---

def create_task(source_filename: str, target_format: str, instructions: str) -> str:
    task_id = str(uuid.uuid4())
    with get_connection() as conn:
        conn.execute(
            "INSERT INTO conversion_tasks (task_id, source_filename, target_format, instructions, status, created_at) VALUES (?, ?, ?, ?, ?, ?)",
            (task_id, source_filename, target_format, instructions, "in_progress", datetime.now(timezone.utc).isoformat())
        )
    return task_id

def update_task(task_id: str, **kwargs):
    """更新任务字段。kwargs: status, execution_code, error_message, token_used, etc."""
    sets = ", ".join(f"{k} = ?" for k in kwargs)
    values = list(kwargs.values()) + [task_id]
    with get_connection() as conn:
        conn.execute(f"UPDATE conversion_tasks SET {sets} WHERE task_id = ?", values)

def get_history(limit: int = 20, offset: int = 0) -> list[dict]:
    with get_connection() as conn:
        rows = conn.execute(
            "SELECT * FROM conversion_tasks ORDER BY created_at DESC LIMIT ? OFFSET ?",
            (limit, offset)
        ).fetchall()
    return [dict(r) for r in rows]

def get_task(task_id: str) -> dict | None:
    with get_connection() as conn:
        row = conn.execute("SELECT * FROM conversion_tasks WHERE task_id = ?", (task_id,)).fetchone()
    return dict(row) if row else None

# --- 技能模板 ---

def save_skill(name: str, description: str, source_schema: dict, target_spec: dict, code: str, column_mapping: dict) -> str:
    skill_id = str(uuid.uuid4())
    now = datetime.now(timezone.utc).isoformat()
    with get_connection() as conn:
        conn.execute(
            "INSERT INTO skill_templates (skill_id, name, description, source_schema, target_spec, code, column_mapping, created_at, updated_at) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)",
            (skill_id, name, description, json.dumps(source_schema, ensure_ascii=False),
             json.dumps(target_spec, ensure_ascii=False), code,
             json.dumps(column_mapping, ensure_ascii=False), now, now)
        )
    return skill_id

def get_skills(limit: int = 20) -> list[dict]:
    with get_connection() as conn:
        rows = conn.execute(
            "SELECT * FROM skill_templates ORDER BY usage_count DESC, created_at DESC LIMIT ?",
            (limit,)
        ).fetchall()
    return [dict(r) for r in rows]

def get_skill(skill_id: str) -> dict | None:
    with get_connection() as conn:
        row = conn.execute("SELECT * FROM skill_templates WHERE skill_id = ?", (skill_id,)).fetchone()
    return dict(row) if row else None

def increment_skill_usage(skill_id: str):
    with get_connection() as conn:
        conn.execute("UPDATE skill_templates SET usage_count = usage_count + 1, updated_at = ? WHERE skill_id = ?",
                     (datetime.now(timezone.utc).isoformat(), skill_id))

def delete_skill(skill_id: str):
    with get_connection() as conn:
        conn.execute("DELETE FROM skill_templates WHERE skill_id = ?", (skill_id,))
```

## 三、技能匹配流程（skill_store.py）

### ChromaDB 初始化

```python
import chromadb
from chromadb.utils import embedding_functions
from app.config import settings

_client = chromadb.PersistentClient(path=settings.chroma_db_dir)
_collection = _client.get_or_create_collection(
    name="skill_schemas",
    embedding_function=embedding_functions.SentenceTransformerEmbeddingFunction(
        model_name="paraphrase-multilingual-MiniLM-L12-v2"  # 支持中文
    )
)
```

### Schema 向量化

每个技能保存时，将以下信息拼接为一段文本，存入 ChromaDB：

```python
def schema_to_text(schema: dict) -> str:
    """将 Schema 转换为可 Embedding 的自然语言文本。"""
    cols = ", ".join(schema["columns"])
    dtypes = ", ".join(f"{k}: {v}" for k, v in schema["dtypes"].items())
    return f"表格包含 {schema['row_count']} 行。列名: {cols}。列类型: {dtypes}。"
```

### 匹配查询

```python
def match_skills(source_schema: dict, top_k: int = 3, threshold: float = 0.7) -> list[dict]:
    """根据源 Schema 匹配最相似的技能模板。

    返回: [{skill_id, skill_name, similarity, suggested_use}, ...]
    """
    query_text = schema_to_text(source_schema)
    results = _collection.query(query_texts=[query_text], n_results=top_k)

    matches = []
    for i, skill_id in enumerate(results["ids"][0]):
        distance = results["distances"][0][i]
        similarity = 1 - distance  # ChromaDB 返回的是距离
        if similarity >= threshold:
            skill = database.get_skill(skill_id)
            if skill:
                matches.append({
                    "skill_id": skill_id,
                    "skill_name": skill["name"],
                    "similarity": round(similarity, 4),
                    "suggested_use": similarity >= 0.85,
                })

    return matches
```

### 完整匹配时机

1. **上传文件后**：`POST /api/match-skills` → 检索相似技能 → 返回匹配列表
2. **用户点击匹配的技能卡片**：加载技能的 `code`，跳过 LLM 推理，直接进入沙箱执行
3. **直接执行失败**：如果历史代码对新数据执行失败，自动回退到 LLM 重新生成

## 四、异常分类与处理策略

```
异常类型层次：

代码错误（可自动修复）
├── 语法错误 → 重试（最多3次）
├── KeyError（列名不匹配） → 重试
├── TypeError（类型错误） → 重试
└── 其他 Python 异常 → 重试（最多3次）

数据异常（需要人工决策）
├── 脏数据（某行数据格式异常） → 挂起 + 推送异常信息
├── 类型不兼容（如字符串无法转数字） → 挂起 + 提供建议
└── 数据截断风险（如 xls 超出 65536 行） → 挂起 + 警告

系统错误（终止）
├── LLM API 调用失败 → 终止 + 错误信息
├── 3次重试耗尽 → 终止 + 错误信息
├── 沙箱超时 → 终止 + 错误信息
└── 文件写入失败 → 终止 + 错误信息
```

## 五、人机协同流程细节

```
Agent 执行中遇到脏数据
     │
     ▼
后端设置 task.status = "awaiting_human_confirmation"
WebSocket 推送: {"type": "blocking", "issues": [...]}
     │
     ▼
前端 confirm-modal.js 弹窗
显示：
  ┌──────────────────────────────────────────────┐
  │  ⚠ 在第 45 行发现异常数据                    │
  │                                              │
  │  列名    当前值          错误                │
  │  ────── ──────────────── ─────────────────── │
  │  员工电话 abc-def-ghij   包含非数字字符       │
  │  建议：清除所有非数字字符 → 01012345678      │
  │                                              │
  │  [采纳建议]  [跳过此行]  [手动修改]  [终止]  │
  └──────────────────────────────────────────────┘
     │
     ├── 采纳建议/手动修改 →
     │     POST /api/confirm-action {
     │       task_id, action: "accept_suggestion",
     │       overrides: {"45": {"员工电话": "01012345678"}}
     │     }
     │     后端：应用修改 → 重新执行受影响的行 → 继续或完成
     │
     ├── 跳过此行 →
     │     POST /api/confirm-action {task_id, action: "skip_row"}
     │     后端：将该行标记为删除 → 继续
     │
     └── 终止 →
           POST /api/confirm-action {task_id, action: "abort"}
           后端：清理任务状态 → 返回失败
```

### confirm-action 后端处理逻辑

```python
def handle_confirm(task_id: str, action: str, overrides: dict | None, websocket_send) -> dict:
    task = tasks[task_id]

    if action == "abort":
        tasks[task_id]["status"] = "cancelled"
        database.update_task(task_id, status="cancelled")
        return {"status": "cancelled"}

    if action == "skip_row":
        # 将问题行从源 df 中删除
        issue_rows = [i["row"] for i in task["pending_issues"]]
        task["source_df"] = task["source_df"].drop(issue_rows)
        # 清空 issues，使用修正后的数据重新执行
        return _retry_execution(task_id, websocket_send)

    if action == "accept_suggestion":
        # 应用建议或用户手动修改的值
        source_df = task["source_df"]
        for row_idx_str, col_updates in (overrides or {}).items():
            row_idx = int(row_idx_str)
            for col, new_val in col_updates.items():
                source_df.at[row_idx, col] = new_val
        task["source_df"] = source_df
        return _retry_execution(task_id, websocket_send)
```
